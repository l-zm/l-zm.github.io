Mysql 设置 unique key 及 联合主键

// 删除约束
alter table `表` drop index `联合主键的名称`;

// 创建 唯一约束主键
alter table `表` add unique key (`字段`)

// 创建 联合主键 联合主键不可重复
alter table `表` add unique key `联合主键的名称` (`字段1`,`字段2`);
